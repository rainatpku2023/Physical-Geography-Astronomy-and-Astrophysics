### 基本假设
+ 辐射部分：太阳辐射功率为$ P_{sol}(t) $，作用在面积$ S $的地面上
+ 生物部分
    - 黑雏菊的生长速率为$ v_1 = v_1(T) $，反照率为$ \alpha_1 = 0.15 $，初始面积为$ S_{10} $，面积为$ S_1(t) $
    - 白雏菊的生长速率为$ v_2 = v_2(T) $，反照率为$ \alpha_2 = 0.65 $，初始面积为$ S_{20} $，面积为$ S_2(t) $
+ 非生物部分
    - 裸露地面的反照率为$ \alpha_2 = 0.4 $，初始面积为$ S_{00} = S-S_{10} - S_{20} $，面积为$ S_0(t) = S - S_1(t) - S_2(t) $
    - 大地的热容为$ C $，辐射出射率为$ \epsilon $，初始温度为$ T_0 $

### 导出方程组
+ 输出辐射功率为：

$ P_{out} = \epsilon \sigma ST^4 $

+ 根据能量守恒：

$ P_{in}-P_{out} = C\cdot\frac{dT}{dt} $

+ 能量输入为：

$ P_{in} = P_{sol}\cdot \frac{(1-\alpha_1) S_1+(1-\alpha_2) S_2+(1-\alpha_0) S_0}{S} $

$ P_{in} = P_{sol} \cdot \frac{(1-\alpha_1) S_1+(1-\alpha_2) S_2+(1-\alpha_0)(S-S_1-S_2)}{S} $

+ 面积生长速率：

$ \frac{dS_1}{dt} = v_1(T) $

$ \frac{dS_2}{dt} = v_2(T) $

+ 综上，得到常微分方程组：

$ \frac{dS_1}{dt} = v_1(T) $

$ \frac{dS_2}{dt} = v_2(T) $

$ \frac{dT}{dt} = \frac{1}{C}[\frac{P_{sol}}{S}((1-\alpha_1) S_1+(1-\alpha_2) S_2+(1-\alpha_0)(S-S_1-S_2))- \epsilon \sigma S T^4] $

+ 定解条件为：

$ S_1(0) = S_{10} $

$ S_2(0) = S_{10} $

$ T(0) = T_{0} $

### 代码实现
[添加绘图模块及字体设置 - 豆包](https://www.doubao.com/thread/wee85db965899aeaf)

```python
import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D
import math

# 设置中文字体，使用常见的SimHei字体
plt.rcParams["font.family"] = ["SimHei"]
plt.rcParams["axes.unicode_minus"] = False  # 解决负号显示问题


# 辐射特性
def P_sol(t):
    return (500 + 500* math.cos(2*math.pi*t/(86400)))  # 大气层顶输入功率随时间变化，可以有更精细的模型

alpha_1 = 0.15  # 黑色雏菊反照率
alpha_2 = 0.65  # 白色雏菊反照率
alpha_0 = 0.4  # 裸地反照率
epsilon = 0.6  # 发射率
sigma = 5.67 * (10 ** (-8))  # 斯特藩-玻尔兹曼常数
T_0 = 273.15 + 21.5  # 初始温度 (K)，21.5摄氏度
C = 42000000  # 热容，相当于10吨水的热容


# 生长特性
def v_1(T):
    # 黑色雏菊面积生长速率随温度变化函数
    return (-1/(86400*365)) * (T - 273.15 - 5) * (T - 273.15 - 25)


def v_2(T):
    # 白色雏菊面积生长速率随温度变化函数
    return (-1/(86400*365)) * (T - 273.15 - 15) * (T - 273.15 - 35)


# 几何特性
S = 1  # 总面积，单位平方米
S_10 = 0.35  # 初始黑色雏菊面积，单位平方米
S_20 = 0.65  # 初始白色雏菊面积，单位平方米

# 模拟相关
init_time = 0 # 模拟开始时间
t_step = 10 # 模拟时间步长，以秒为单位
total_time = 86400*75 # 总模拟时间，以秒为单位
epoch = int(total_time / t_step) # 总的迭代次数
time_array = [init_time]  # 时间戳列表
S_1_array = [S_10]  # 黑色雏菊面积变化列表
S_2_array = [S_20]  # 白色雏菊面积变化列表
T_array = [T_0]  # 温度变化列表

# 开始模拟
t = init_time
for i in range(epoch):
    S_1 = S_1_array[-1]
    S_2 = S_2_array[-1]
    T = T_array[-1]

    # 计算新的雏菊面积，确保面积在合理范围内 [0, 1]
    new_S_1 = max(0, min(S_1 + v_1(T) * t_step, 1))
    new_S_2 = max(0, min(S_2 + v_2(T) * t_step, 1))

    # 确保总面积不超过1
    if new_S_1 + new_S_2 > 1:
        ratio = 1 / (new_S_1 + new_S_2)
        new_S_1 *= ratio
        new_S_2 *= ratio

    # 计算新的温度
    new_T = T + (t_step / C) * (
        (P_sol(t) / S) * ((1 - alpha_1) * S_1 + (1 - alpha_2) * S_2 + (1 - alpha_0) * (S - S_1 - S_2))
        - epsilon * sigma* S * (T ** 4)
    )

    # 更新时间
    t += t_step

    # 将新值添加到数组
    S_1_array.append(new_S_1)
    S_2_array.append(new_S_2)
    T_array.append(new_T)
    time_array.append(t)

# 绘图
fig = plt.figure(figsize=(15, 10))

# 3D 轨迹图 - S1, S2, T 空间中的运动轨迹
ax1 = fig.add_subplot(221, projection='3d')
ax1.plot(S_1_array, S_2_array, T_array, linewidth=0.5, alpha=0.8)
ax1.set_xlabel('黑色雏菊面积 (S_1)')
ax1.set_ylabel('白色雏菊面积 (S_2)')
ax1.set_zlabel('温度 (T, K)')
ax1.set_title('系统在S_1-S_2-T空间中的运动轨迹')

# 黑色雏菊面积随时间变化
ax2 = fig.add_subplot(222)
ax2.plot(time_array, S_1_array)
ax2.set_xlabel('时间')
ax2.set_ylabel('黑色雏菊面积 (S_1)')
ax2.set_title('黑色雏菊面积随时间变化')

# 白色雏菊面积随时间变化
ax3 = fig.add_subplot(223)
ax3.plot(time_array, S_2_array, 'orange')
ax3.set_xlabel('时间')
ax3.set_ylabel('白色雏菊面积 (S_2)')
ax3.set_title('白色雏菊面积随时间变化')

# 温度随时间变化
ax4 = fig.add_subplot(224)
ax4.plot(time_array, T_array, 'red')
ax4.set_xlabel('时间')
ax4.set_ylabel('温度 (T, K)')
ax4.set_title('温度随时间变化')

plt.tight_layout()
plt.show()
```
