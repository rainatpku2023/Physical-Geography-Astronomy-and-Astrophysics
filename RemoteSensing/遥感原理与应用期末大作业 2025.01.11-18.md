//Get ARVI Image for "LANDSAT/LT04/C02/T1"(1983-1992) "LANDSAT/LT05/C02/T1"(1985-2000) "LANDSAT/LE07/C02/T1"(2000-2024)
function getArviImage1(image){
  var gamma = 1
  image = image.unmask()
  var rhoRB = image.unmask().expression("rhoR-Gamma*(rhoB-rhoR)",{rhoR:image.select(["B3"]), //注意写法
                                                         Gamma:gamma, //gamma
                                                         rhoB:image.select(["B1"])})
    .rename("rhoRB")
  image = image.unmask().addBands(rhoRB)
  var arviImage = image.unmask().normalizedDifference(["B4","rhoRB"]);
  return arviImage
}

//Get ARVI Image for "LANDSAT/LC08/C02/T1"(2014-2024) "LANDSAT/LC08/C02/T1"(2022-2024)
function getArviImage2(image){
  var gamma = 1
  image = image.unmask()
  var rhoRB = image.expression("rhoR-Gamma*(rhoB-rhoR)",{rhoR:image.select(["B4"]), //注意写法
                                                         Gamma:gamma, //gamma
                                                         rhoB:image.select(["B2"])})
    .rename("rhoRB")
  image = image.unmask().addBands(rhoRB)
  var arviImage = image.unmask().normalizedDifference(["B5","rhoRB"]);
  return arviImage
}

//Get Mean ARVI Image.
function getMeanARVIImage(startDate,endDate,dataSeriesName){
  var rawImageCollection = ee.ImageCollection(dataSeriesName)
    .filterDate(String(startDate),String(endDate));
  if (dataSeriesName === "LANDSAT/LT04/C02/T1" ||
      dataSeriesName === "LANDSAT/LT05/C02/T1" ||
      dataSeriesName === "LANDSAT/LE07/C02/T1"){
    var arviImageCollection = rawImageCollection.map(getArviImage1)
  }else if(dataSeriesName === "LANDSAT/LC08/C02/T1" ||
           dataSeriesName === "LANDSAT/LC09/C02/T1"){
    var arviImageCollection = rawImageCollection.map(getArviImage2)
  }
  var meanArviImage = arviImageCollection.mean();
  return meanArviImage
}

//Get Mean ARVI for Image.
function getMeanValueForImage(image,frame){
  var meanValue = image.unmask().reduceRegion({
    reducer: ee.Reducer.mean(),
    geometry: frame,
    scale:30,
    maxPixels: 1e20
  })
  return meanValue
}

def input_message(length_of_message):
    output_list = []
    for i in range(length_of_message):
        input_line = input()
        input_line = input_line[len(str(i))+1:] # 删除前缀
        if i == 0: # Name of data series
            input_line = input_line
        elif i == 1: # Index of region, 0-6
            input_line = int(input_line)
        elif i == 2 or i == 3: # Starting year and ending year
            input_line = int(input_line)
        else: # Data
            input_line = input_line[2:-1] # 4: [0.1283212641305224] 冒号后面竟然还有个空格
        output_list.append(input_line)
    print("Here are the results!")
    for i in range(length_of_message):
        print(output_list[i])

//Get ARVI Image for landsat TM.
function getArviImage(image,gamma){
  var rhoRBImage = image.expression("rhoR-Gamma*(rhoB-rhoR)",{rhoR:image.select["B3"],
                                                         Gamma:gamma,
                                                         rhoB:image.select["B1"]})
    .rename("rhoRB")
  image = image.addBands(rhoRBImage)
  var arviImage = image.normalizedDifference(["rhoRB","B4"]);
  return arviImage
}

//Get Mean ARVI Image.
function getMeanARVI(startDate,endDate,imageCollectionName,gamma){
  var rawImageCollection = ee.ImageCollection(imageCollectionName)
    .filterDate(String(startDate),String(endDate));

  var arviImageCollection = rawImageCollection.map(getArviImage,gamma);
  var meanArviImage = arviImageCollection.mean();
  
  return meanArviImage
}
